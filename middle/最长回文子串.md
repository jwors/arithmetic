```javascript
 * @param {string} s

var longestPalindrome = function (s) {
  /*
    1. 中心轴对称 长度最小为2
    2. 如果是偶数 那么 length / 2 两边相等
    3. 如果是奇数 中间数长度为1，
  */
  if (s.length === 2) {
    const status = s.slice(0, 1) === s.slice(1, 2);
    if (status) {
      return s;
    }
  }
  let startIndex = 0;
  let endIndex = 1;
  let st = s.slice(0, 1);
  for (let k = 0; k < s.length - 1; k++) {
    for (let i = k + 1; i <= s.length; i++) {
      let str = s.slice(k, i);
      if (str.length % 2 === 0) {
        // 偶数
        const length = str.length / 2;
        if (
          str.slice(0, length) ===
          str
            .slice(length, length * 2)
            .split('')
            .reverse()
            .join('')
        ) {
          if (st === null || st.length < str.length) {
            st = str;
          }
        }
      } else {
        //奇数
        const length = (str.length - 1) / 2;
        if (
          str.slice(0, length) ===
          str
            .slice(length + 1, str.length)
            .split('')
            .reverse()
            .join('')
        ) {
          if (st === null || st.length < str.length) {
            st = str;
          }
        }
      }
    }
  }
  console.log(st);
  return st;
};
```

```javascript
/**
 * @param {string} s
 * @return {string}
 */
var longestPalindrome = function (s) {
  if (s.length < 2) {
    return s;
  }
  let res = '';
  for (let i = 0; i < s.length; i++) {
    // 回文子串长度是奇数
    helper(i, i);
    // 回文子串长度是偶数
    helper(i, i + 1);
  }

  function helper(m, n) {
    while (m >= 0 && n < s.length && s[m] == s[n]) {
      m--;
      n++;
    }
    // 注意此处m,n的值循环完后  是恰好不满足循环条件的时刻
    // 此时m到n的距离为n-m+1，但是mn两个边界不能取 所以应该取m+1到n-1的区间  长度是n-m-1
    if (n - m - 1 > res.length) {
      // slice也要取[m+1,n-1]这个区间
      res = s.slice(m + 1, n);
    }
  }
  return res;
};
```
